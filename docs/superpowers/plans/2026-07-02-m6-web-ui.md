```typescript
// tools.ts
import type { CronJobPort } from "../cron/ports.js";
import type { CronScheduler } from "../cron/scheduler.js";

export function createUpdateLoopConfigTool(
  cronJobId: string,
  port: CronJobPort,
  scheduler: CronScheduler,
) {
  return {
    name: "update_loop_config",
    description: "Set the cron schedule for this Loop",
    inputSchema: {
      type: "object",
      properties: {
        cronExpr: { type: "string", description: "5-field cron expression" },
      },
      required: ["cronExpr"],
    },
    execute: async (input: { cronExpr: string }) => {
      // Validate
      const parts = input.cronExpr.trim().split(/\s+/);
      if (parts.length !== 5) throw new Error("Must be 5-field cron expression");

      // Update DB
      port.updateCronJob(cronJobId, {
        cronExpr: input.cronExpr,
        updatedAt: Date.now(),
      });

      // Re-register scheduler
      const job = port.getCronJob(cronJobId);
      if (job) scheduler.register(job);
    },
  };
}
```

```typescript
.post("/api/loops", async ({ body, set }) => {
  const loopId = idGen();
  const loopPath = `loops/${loopName}`;
  
  // 1. Create cron_job row
  const job = await cronSvc.createCronJob({ ...agentId: "loop-agent", cronExpr: "", loopConfigPath: loopPath });
  
  // 2. Create Conversation
  convPort.createConversation({ conversationId: job.cronJobId, title: body.name, origin: "loop" });
  convPort.addMember({ conversationId: job.cronJobId, memberId: "owner", kind: "agent", agentId: "loop-agent" });
  
  // 3. Create directory
  await mkdir(`${dataDir}/${loopPath}/skills`, { recursive: true });
  
  // 4. Run AgentSession
  const session = sessionFactory.getOrCreate(`loop:create:${loopId}`, buildSpec({ ... }));
  await sessionFactory.enqueuePrompt(`loop:create:${loopId}`, promptWithIntent);
  
  // 5. Read LOOP.md → return preview
  const loopMd = await Bun.file(`${dataDir}/${loopPath}/LOOP.md`).text();
  const cfg = parseLoopConfig(loopMd);
  return { loop: { id: job.cronJobId, ... }, preview: loopMd };
})
```
