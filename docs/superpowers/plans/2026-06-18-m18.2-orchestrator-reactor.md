# M18.2 Orchestrator Reactor Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the Orchestrator reactor — a transfer-table-driven mechanism that auto-advances Issues through their lifecycle by listening to run completions and kicking off the next agent.

**Architecture:** Move the TRANSITIONS table from issue/entities.ts to a new orchestrator/transitions.ts with added agentId + promptTemplate fields. Add an issue_id column to run_origin for linking runs to Issues (and as the @mention isolation discriminant). Build createOrchestrator() factory with startStep (construct spec from transfer table, validate agent, write run_origin) and onRunComplete (listen to run completions, call applyTransition, start next step). Wire via supervisor.onRunComplete + onIssueCreated callback injection. Dependency direction: IssueStatus pure type stays in issue; transitions + runtime in orchestrator; issue depends on orchestrator for LEGAL_TRANSITIONS; orchestrator depends on issue for types and applyTransition; no runtime cycle.

**Tech Stack:** TypeScript (ES2023), bun:sqlite, Zod, bun:test, factory functions (no classes except Error subclasses and RuntimeOpsStore)

## Spec Corrections (Grill)

Findings from comparing spec §6 against actual code:

1. **Migration name collision**: Spec says `events_v9_run_origin_issue` but v9 already exists (`events_v9_runner_health`, id 3008). Must use `events_v13_run_origin_issue`, id **3012**.

2. **`agentSvc.getById` throws on archived**: agent/service.ts line 50: `if (!row || row.archivedAt) throw new AgentNotFoundError(id)`. Spec's check `agent.archivedAt != null` after `.catch(() => null)` is dead code. Plan uses simpler `if (!agent)` after catch.

3. ~~P6 over-engineered~~ **REVERSED after code verification**: `issue.threadId` is a real composite execution thread identifier (e.g. `convId:memberId`), NOT a bare ULID. `parseThreadId` WILL resolve it with `cid` non-empty, so the `if(!cid) return` early exit does NOT fire. Without the `run_origin.issueId` check in P6, conversation's `onRunComplete` would enter Phase 3 and trigger @mention for issue-driven runs. **P6's `opsStore` parameter is necessary — plan restores it.**

4. **`IssueService` type not exported**: issue/index.ts doesn't export `IssueService`. Plan adds it so orchestrator can import cleanly.

5. **`RunOriginRow` all required fields**: Must make `issueId` optional. Plan adds `issueId?: string | null`.

6. **`insertRunOrigin` uses 9 positional params**: Must expand to 10 (with issue_id). Plan updates both the SQL and the RUN_ORIGIN_COLS constant.

```ts
// apps/backend/src/features/orchestrator/transitions.ts
import type { IssueStatus } from "../issue/entities.js";

export interface Transition {
  from: IssueStatus;
  to: IssueStatus;
  agentId: string;
  promptTemplate: string;
}

export const TRANSITIONS = [
  {
    from: "planned",
    to: "in_progress",
    agentId: "planner",
    promptTemplate: "为 Issue「{{title}}」制定开发计划并开始实现。",
  },
  {
    from: "in_progress",
    to: "in_review",
    agentId: "developer",
    promptTemplate: "完成 Issue「{{title}}」的实现，提交待 Review。",
  },
  {
    from: "in_review",
    to: "done",
    agentId: "reviewer",
    promptTemplate: "Review Issue「{{title}}」的实现，通过则标记完成。",
  },
] as const satisfies ReadonlyArray<Transition>;

export function deriveStatuses(table: ReadonlyArray<Transition>): IssueStatus[] {
  const seen = new Set<IssueStatus>();
  for (const t of table) {
    seen.add(t.from);
    seen.add(t.to);
  }
  return [...seen];
}

export function deriveLegalMap(
  table: ReadonlyArray<Transition>,
): Record<IssueStatus, IssueStatus[]> {
  const map = Object.fromEntries(deriveStatuses(table).map((s) => [s, [] as IssueStatus[]]));
  for (const t of table) map[t.from]!.push(t.to);
  return map as Record<IssueStatus, IssueStatus[]>;
}

export function nextTransition(
  table: ReadonlyArray<Transition>,
  from: IssueStatus,
): Transition | undefined {
  return table.find((t) => t.from === from);
}

export const ISSUE_STATUSES: readonly IssueStatus[] = deriveStatuses(TRANSITIONS);
export const LEGAL_TRANSITIONS: Readonly<Record<IssueStatus, IssueStatus[]>> =
  deriveLegalMap(TRANSITIONS);
```

```ts
// apps/backend/src/features/orchestrator/render.ts
/** Only {{var}} string interpolation. Missing vars fall back to empty string.
 *  Deliberately no conditionals, loops, or filters — no template DSL. */
export function renderPrompt(template: string, vars: Record<string, string>): string {
  return template.replace(/\{\{(\w+)\}\}/g, (_, k) => vars[k] ?? "");
}
```

```ts
// apps/backend/src/features/orchestrator/reactor.ts
import type { IssueRow, IssueStatus } from "../issue/entities.js";
import type { IssueService } from "../issue/service.js";
import type { AgentService } from "../agent/service.js";
import type { RunSupervisor } from "../run/supervisor.js";
import type { RuntimeOpsStore } from "../runtime-ops/store.js";
import { renderPrompt } from "./render.js";
import { nextTransition, TRANSITIONS } from "./transitions.js";

export class OrchestratorAgentMissingError extends Error {
  constructor(agentId: string, issueId: string) {
    super(`Orchestrator: agent not found or archived: ${agentId} (issue ${issueId})`);
    this.name = "OrchestratorAgentMissingError";
  }
}

export interface OrchestratorDeps {
  issueSvc: IssueService;
  agentSvc: AgentService;
  supervisor: RunSupervisor;
  opsStore: RuntimeOpsStore;
  buildSpec: (agentId: string, threadId: string, input: string) => Promise<Record<string, unknown>>;
  idGen: () => string;
  now?: () => number;
}

export function createOrchestrator(deps: OrchestratorDeps) {
  const { issueSvc, agentSvc, supervisor, opsStore, buildSpec, idGen } = deps;

  async function startStep(issue: IssueRow): Promise<{ runId: string } | null> {
    const t = nextTransition(TRANSITIONS, issue.status);
    if (!t) return null;

    // getById throws if missing or archived — catch both
    const agent = await agentSvc.getById(t.agentId).catch(() => null);
    if (!agent) {
      throw new OrchestratorAgentMissingError(t.agentId, issue.issueId);
    }

    const runId = idGen();
    const prompt = renderPrompt(t.promptTemplate, { title: issue.title, issueId: issue.issueId });
    const spec = await buildSpec(t.agentId, issue.threadId, prompt);

    await supervisor.startMainRun(runId, issue.threadId, spec);

    opsStore.insertRunOrigin({
      runId,
      issueId: issue.issueId,
      conversationId: "",
      sourceLedgerSeq: 0,
      agentMemberId: t.agentId,
      surface: "orchestrator",
      traceId: "",
      traceparent: "",
      idempotencyKey: `issue:${issue.issueId}:${issue.status}:run`,
      createdAt: (deps.now ?? Date.now)(),
    });

    return { runId };
  }

  async function onRunComplete(
    threadId: string,
    runId: string,
    status: string,
    kind: string,
  ): Promise<void> {
    if (kind === "reflect") return;

    const origin = opsStore.getRunOrigin(runId);
    const issueId = origin?.issueId;
    if (!issueId) return;

    if (status !== "succeeded") {
      console.warn(`[orchestrator] run ${runId} for issue ${issueId} ended ${status}; not advancing`);
      return;
    }

    const issue = issueSvc.port.getIssue(issueId);
    if (!issue) return;

    const t = nextTransition(TRANSITIONS, issue.status);
    if (!t) return;

    let advanced: IssueRow;
    try {
      advanced = issueSvc.applyTransition(issueId, t.to);
    } catch (err) {
      console.warn(`[orchestrator] applyTransition skipped for ${issueId}: ${String(err)}`);
      return;
    }

    try {
      await startStep(advanced);
    } catch (err) {
      console.error(`[orchestrator] failed to start next step for ${issueId}: ${String(err)}`);
    }
  }

  return { startStep, onRunComplete };
}
```
