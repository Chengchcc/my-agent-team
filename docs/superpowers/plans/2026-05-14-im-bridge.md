import { existsSync, readFileSync, mkdirSync, writeFileSync, readdirSync } from 'node:fs';
import { join } from 'node:path';
import { parse as parseYaml } from 'yaml';
import { z } from 'zod';
import type { AgentProfile, BotsConfig } from './types';
import { homedir } from 'node:os';

const agentProfileSchema = z.object({
  id: z.string(),
  workspace: z.string(),
  model: z.string().optional(),
  toolProfile: z.enum(['read_only', 'code_editor', 'general']),
  workingDir: z.string(),
  allowedRoots: z.array(z.string()).optional(),
  permissionTimeoutMs: z.number().optional(),
});

const botsConfigSchema = z.object({
  profiles: z.record(agentProfileSchema),
  bots: z.array(z.object({
    larkAppId: z.string(),
    larkAppSecret: z.string(),
    profileId: z.string(),
    allowedUsers: z.array(z.string()).optional(),
  })),
});

export function resolvePath(p: string): string {
  return p.replace(/^~/, () => homedir() ?? '/root');
}

export function getBotsConfigPath(): string {
  return join(homedir() ?? '/root', '.my-agent', 'bots.yml');
}

export function loadBotsConfig(): BotsConfig {
  const configPath = getBotsConfigPath();
  if (!existsSync(configPath)) {
    throw new Error(`Bots config not found at ${configPath}. Run "my-agent bot add" first.`);
  }
  const raw = readFileSync(configPath, 'utf-8');
  const parsed = parseYaml(raw);
  const result = botsConfigSchema.safeParse(parsed);
  if (!result.success) {
    throw new Error(`Invalid bots config: ${result.error.message}`);
  }
  return result.data;
}

export function getProfile(profileId: string): AgentProfile {
  const config = loadBotsConfig();
  const profile = config.profiles[profileId];
  if (!profile) {
    throw new Error(`Profile "${profileId}" not found in bots.yml`);
  }
  return profile;
}

export function getBot(larkAppId: string) {
  const config = loadBotsConfig();
  const bot = config.bots.find(b => b.larkAppId === larkAppId);
  if (!bot) throw new Error(`Bot ${larkAppId} not found`);
  return { config: bot, profile: getProfile(bot.profileId) };
}

export function listProfiles(): AgentProfile[] {
  return Object.values(loadBotsConfig().profiles);
}

// src/im/lark/client.ts
import { Client, LoggerLevel } from '@larksuiteoapi/node-sdk';

let _client: InstanceType<typeof Client> | null = null;
let _appId = '';
let _appSecret = '';

export function initLarkClient(appId: string, appSecret: string): void {
  _appId = appId;
  _appSecret = appSecret;
  _client = new Client({
    appId,
    appSecret,
    loggerLevel: process.env.DEBUG ? LoggerLevel.info : LoggerLevel.warn,
  });
}

function client(): InstanceType<typeof Client> {
  if (!_client) throw new Error('Lark client not initialized');
  return _client;
}

export async function sendMessage(
  chatId: string, content: string, msgType: string = 'text',
): Promise<string> {
  const c = client();
  const body = msgType === 'text' ? JSON.stringify({ text: content }) : content;
  const res = await (c.im.v1.message as any).create({
    params: { receive_id_type: 'chat_id' },
    data: { receive_id: chatId, msg_type: msgType as any, content: body },
  });
  if (res.code !== 0) throw new Error(`sendMessage failed: ${res.msg} (code: ${res.code})`);
  return res.data?.message_id ?? '';
}

export async function replyMessage(
  messageId: string, content: string, msgType: string = 'text',
  replyInThread: boolean = false,
): Promise<string> {
  const c = client();
  const body = msgType === 'text' ? JSON.stringify({ text: content }) : content;
  const res = await (c.im.v1.message as any).reply({
    path: { message_id: messageId },
    data: {
      msg_type: msgType as any,
      content: body,
      ...(replyInThread ? { reply_in_thread: true } : {}),
    },
  });
  if (res.code !== 0) throw new Error(`replyMessage failed: ${res.msg} (code: ${res.code})`);
  return res.data?.message_id ?? '';
}

export async function updateMessage(

  const c = client();

export async function getChatInfo(
  chatId: string,
): Promise<{ userCount: number; botCount: number }> {
  const c = client();
  const res = await (c.im.v1.chat as any).get({ path: { chat_id: chatId } });

const chatModeCache = new Map<string, { mode: 'group' | 'topic' | 'p2p'; cachedAt: number }>();
const CHAT_MODE_TTL = 5 * 60_000;

export async function getChatMode(
  chatId: string, opts: { forceRefresh?: boolean } = {},
): Promise<'group' | 'topic' | 'p2p'> {
  const key = `${_appId}::${chatId}`;
  const cached = chatModeCache.get(key);
  if (!opts.forceRefresh && cached && Date.now() - cached.cachedAt < CHAT_MODE_TTL) {
    return cached.mode;
  }
  let mode: 'group' | 'topic' | 'p2p' = 'group';
  try {
    const c = client();
    const res = await (c.im.v1.chat as any).get({ path: { chat_id: chatId } });
    if (res.code === 0) {
      const rawMode = String(res.data?.chat_mode ?? '').toLowerCase();
      const rawType = String(res.data?.chat_type ?? '').toLowerCase();
      const rawGmt = String(res.data?.group_message_type ?? '').toLowerCase();
      if (rawType === 'p2p') mode = 'p2p';
      else if (rawMode === 'topic' || rawGmt === 'thread') mode = 'topic';
      else mode = 'group';
    }
  } catch { /* fallback to 'group' */ }
  chatModeCache.set(key, { mode, cachedAt: Date.now() });
  return mode;
}

export async function getBotOpenId(): Promise<{ openId: string; name: string }> {
  const tokenRes = await fetch(
    'https://open.feishu.cn/open-apis/auth/v3/tenant_access_token/internal',
    {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ app_id: _appId, app_secret: _appSecret }),
    },
  );
  const tokenData = await tokenRes.json() as any;
  if (tokenData.code !== 0) throw new Error(`Failed to get token: ${tokenData.msg}`);

  const botRes = await fetch('https://open.feishu.cn/open-apis/bot/v3/info/', {
    headers: { Authorization: `Bearer ${tokenData.tenant_access_token}` },
  });
  const botData = await botRes.json() as any;
  if (botData.code !== 0) throw new Error(`Failed to get bot info: ${botData.msg}`);
  return { openId: botData.bot?.open_id, name: botData.bot?.app_name ?? '' };
}

export async function getMessageDetail(messageId: string): Promise<any> {
  const c = client();
  const res = await (c.im.v1.message as any).get({
    path: { message_id: messageId },
    params: { card_msg_content_type: 'user_card_content' },
  });
  if (res.code !== 0) throw new Error(`getMessageDetail failed: ${res.msg}`);
  return res.data;
}

export async function downloadResource(
  messageId: string, fileKey: string, type: 'image' | 'file', savePath: string,
): Promise<void> {
  const c = client();
  const res = await (c.im.v1.messageResource as any).get({
    path: { message_id: messageId, file_key: fileKey },
    params: { type },
  });
  const { writeFileSync, mkdirSync, existsSync } = require('node:fs');
  const { dirname } = require('node:path');
  const dir = dirname(savePath);
  if (!existsSync(dir)) mkdirSync(dir, { recursive: true });
  if (res instanceof Buffer) {
    writeFileSync(savePath, res);
  } else if (res && typeof res === 'object' && 'writeFile' in res) {
    await res.writeFile(savePath);
  } else {
    const chunks: Buffer[] = [];
    for await (const chunk of res as AsyncIterable<Buffer>) {
      chunks.push(Buffer.from(chunk));
    }
    writeFileSync(savePath, Buffer.concat(chunks));
  }
}

// src/im/lark/message-parser.ts

export interface ParsedMessage {
  messageId: string;
  chatId: string;
  rootId?: string;
  threadId?: string;
  senderId?: string;
  senderType: 'user' | 'app' | 'unknown';
  chatType: 'group' | 'p2p';
  msgType: string;
  content: string;
  mentions: LarkMention[];
  createTime: string;
}

export interface LarkMention {
  key: string;
  name: string;
  openId?: string;
}

export function parseEventMessage(data: any): ParsedMessage {
  const message = data.message ?? data;
  const sender = data.sender;

  const mentions: LarkMention[] = (message.mentions ?? []).map((m: any) => ({
    key: m.key ?? '',
    name: m.name ?? '',
    openId: m.id?.open_id,
  }));

  let content = '';
  const msgType: string = message.msg_type ?? 'text';
  try {
    const obj = JSON.parse(message.content ?? '{}');
    if (msgType === 'text' && typeof obj.text === 'string') {
      content = obj.text;
    } else {
      // post type
      const inner = obj.zh_cn ?? obj.en_us ?? obj;
      if (Array.isArray(inner?.content)) {
        const parts: string[] = [];
        for (const para of inner.content) {
          if (!Array.isArray(para)) continue;
          for (const node of para) {
            if (node?.tag === 'text' && typeof node.text === 'string') {
              parts.push(node.text);
            }
          }
        }
        content = parts.join('');
      }
    }
  } catch { /* empty */ }

  // Resolve @mention placeholders (@_user_1 → @Name)
  for (const m of mentions) {
    if (m.key && m.name) {
      content = content.split(m.key).join(`@${m.name}`);
    }
  }

  return {
    messageId: message.message_id ?? '',
    chatId: message.chat_id ?? '',
    rootId: message.root_id,
    threadId: message.thread_id,
    senderId: sender?.sender_id?.open_id,
    senderType: sender?.sender_type === 'app' ? 'app' : 'user',
    chatType: message.chat_type === 'p2p' ? 'p2p' : 'group',
    msgType,
    content: content.trim(),
    mentions,
    createTime: message.create_time ?? '',
  };
}

export function stripLeadingMentions(content: string, mentions: LarkMention[]): string {
  let result = content;
  for (const m of mentions) {
    const atName = `@${m.name}`;
    if (result.startsWith(atName)) {
      result = result.slice(atName.length).trimStart();
    }
  }
  return result;
}

// src/im/lark/card-builder.ts

function escapeMd(s: string): string {
  return s.replace(/[*_~`\[\]\\]/g, c => `\\${c}`);
}

const statusMap: Record<string, { label: string; template: string }> = {
  starting: { label: '启动中…', template: 'yellow' },
  working: { label: '工作中', template: 'blue' },
  idle: { label: '等待输入', template: 'green' },
  analyzing: { label: '正在分析…', template: 'purple' },
  error: { label: '出错', template: 'red' },
};

export function buildStreamingCard(params: {
  sessionId: string;
  rootId: string;
  title: string;
  markdownContent: string;
  status: 'starting' | 'working' | 'idle' | 'analyzing' | 'error';
  displayMode: 'hidden' | 'markdown';
  cardNonce?: string;
}): string {
  const { sessionId, rootId, title, markdownContent, status, displayMode, cardNonce } = params;
  const st = statusMap[status] ?? statusMap.working;
  const elements: any[] = [];

  if (displayMode === 'markdown' && markdownContent) {
    const truncated = markdownContent.length > 3000
      ? markdownContent.slice(0, 3000) + '\n\n_(输出已截断)_'
      : markdownContent;
    elements.push({
      tag: 'markdown',
      content: truncated,
    });
    elements.push({ tag: 'hr' });
  }

  const headerActions: any[] = [
    {
      tag: 'button',
      text: { tag: 'plain_text', content: displayMode === 'hidden' ? '📖 显示输出' : '📕 隐藏输出' },
      type: 'default',
      value: { action: 'toggle_display', root_id: rootId, session_id: sessionId, ...(cardNonce ? { card_nonce: cardNonce } : {}) },
    },
    {
      tag: 'button',
      text: { tag: 'plain_text', content: '🔄 重启' },
      type: 'default',
      value: { action: 'restart', root_id: rootId, session_id: sessionId },
    },
    {
      tag: 'button',
      text: { tag: 'plain_text', content: '❌ 关闭会话' },
      type: 'danger',
      value: { action: 'close', root_id: rootId, session_id: sessionId },
    },
  ];
  elements.push({ tag: 'action', actions: headerActions });

  const card = {
    config: { wide_screen_mode: true },
    header: {
      title: { tag: 'plain_text', content: `🤖 ${escapeMd(title)} — ${st.label}` },
      template: st.template,
    },
    elements,
  };
  return JSON.stringify(card);
}

  provider: Provider;
  toolRegistry: ToolRegistry;
  profile: AgentProfile;
  larkAppId: string;
  memoryStoreFactory: (profileId: string) => Promise<MemoryStore>;
  sessionStore: SessionStore;

export class SessionManager {
  private sessions = new Map<string, DaemonSession>();
  private agents = new Map<string, Agent>();
  private contextManagers = new Map<string, ContextManager>();
  private deps: SessionManagerDeps;

  constructor(deps: SessionManagerDeps) {
    this.deps = deps;
  }

  getSession(key: string): DaemonSession | undefined {
    return this.sessions.get(key);
  }

  listSessions(): DaemonSession[] {
    return [...this.sessions.values()];
  }

  async createSession(
    ctx: RoutingContext,
    initialPrompt: string,
    ownerOpenId?: string,
  ): Promise<DaemonSession> {
    const { profile, larkAppId, sessionStore, toolRegistry } = this.deps;
    const key = sessionKey(ctx.anchor, larkAppId);

    // Create isolated ContextManager
    const contextManager = new ContextManager({
      tokenLimit: 200_000,
      defaultSystemPrompt: '', // will be set per-turn
    });

    // Build filtered tool registry using profile.toolProfile
    const subToolRegistry = new ToolRegistry();
    const profileTools = PROFILE_TOOLS[profile.toolProfile];
    const useProfileFilter = profileTools.length > 0;

    for (const toolDef of toolRegistry.getAllDefinitions()) {
      if (ALWAYS_EXCLUDE.has(toolDef.name)) continue;
      if (toolDef.name.startsWith('Task')) continue;
      if (useProfileFilter && !profileTools.includes(toolDef.name)) continue;
      const impl = toolRegistry.get(toolDef.name);
      if (impl) subToolRegistry.register({ getDefinition: () => toolDef, execute: (p, c) => impl.execute(p, c) });
    }

    // Create Agent instance
    const agent = new Agent({
      provider: this.deps.provider,
      contextManager,
      config: {
        cwd: profile.workingDir,
        tokenLimit: 200_000,
        model: profile.model,
        ...(profile.allowedRoots ? { allowedRoots: profile.allowedRoots } : {}),
      },
      toolRegistry: subToolRegistry,
    });

    // Create session record
    const session = sessionStore.createSession(
      ctx.chatId,
      ctx.scope === 'thread' ? ctx.anchor : ctx.messageId,
      initialPrompt.substring(0, 50),
      ctx.chatType,
    );
    session.larkAppId = larkAppId;

    const ds: DaemonSession = {
      session,
      larkAppId,
      chatId: ctx.chatId,
      chatType: ctx.chatType,
      scope: ctx.scope,
      spawnedAt: Date.now(),
      lastMessageAt: Date.now(),
      hasHistory: false,
      workingDir: profile.workingDir,
      pendingPrompt: initialPrompt,
      busy: false,
      messageQueue: [],
      ownerOpenId: ownerOpenId,
    };

    this.sessions.set(key, ds);
    this.agents.set(key, agent);
    this.contextManagers.set(key, contextManager);

    return ds;
  }

  async runAgentTurn(
    ds: DaemonSession,
    prompt: string,
  ): Promise<void> {
    const key = sessionKey(sessionAnchorId(ds), ds.larkAppId);
    const agent = this.agents.get(key);
    if (!agent) throw new Error(`Agent not found for session ${key}`);

    ds.busy = true;
    try {
      const loopConfig = {
        maxTurns: 25,
        timeoutMs: 600_000,
      };

      for await (const event of agent.runAgentLoop(
        { role: 'user', content: prompt },
        loopConfig,
      )) {
        this.deps.onAgentEvent(key, event);
      }
    } finally {
      ds.busy = false;
      // Process queue
      if (ds.messageQueue.length > 0) {
        const next = ds.messageQueue.shift();
        if (next) await this.runAgentTurn(ds, next);
      }
    }
  }

  queueMessage(ds: DaemonSession, content: string): void {
    if (ds.busy) {
      ds.messageQueue.push(content);
    }
  }

  removeSession(key: string): void {
    this.sessions.delete(key);
    this.agents.delete(key);
    this.contextManagers.delete(key);
  }

  getAgent(key: string): Agent | undefined {
    return this.agents.get(key);
  }

  get activeCount(): number {
    return this.sessions.size;
  }
}

  handleThreadReply: (data: any, ctx: RoutingContext) => Promise<void>;
  handleCardAction: (data: any) => Promise<any>;

export function startLarkEventDispatcher(
  larkAppId: string,
  larkAppSecret: string,
  handlers: EventHandlers,
  botOpenId: string,
): Lark.WSClient {
  const eventDispatcher = new Lark.EventDispatcher({}).register({
    'card.action.trigger': async (data: any) => {
      try {
        const cardBody = await handlers.handleCardAction(data);
        if (cardBody) return { card: { type: 'raw', data: cardBody } };
      } catch (err) {
        logger.error(`Card action error: ${err}`);
      }
      return undefined;
    },
    'im.message.receive_v1': async (data: any) => {
      try {
        const message = data.message;
        const sender = data.sender;
        if (!message) return;

        // Skip self messages except /close
        if (sender?.sender_type === 'app') {
          const isSelf = sender.sender_id?.open_id === botOpenId;
          if (isSelf) {
            try {
              const body = JSON.parse(message.content ?? '{}');
              if (body.text?.trim() !== '/close') return;
            } catch { return; }
          } else {
            // Foreign bot @mention — only route if we're mentioned
            const mentions = message.mentions ?? [];
            if (!mentions.some((m: any) => m.id?.open_id === botOpenId)) return;
          }
        }

        const { parsed } = parseEventMessage(data);
        // parseEventMessage returns the ParsedMessage directly
        const msg = parseEventMessage(data);
        const chatId = msg.chatId;
        const chatType = msg.chatType;
        const messageId = msg.messageId;
        const rootId = msg.rootId;
        const threadId = msg.threadId;
        const senderOpenId = msg.senderId;

        // Decide routing
        const routing = await decideRouting(larkAppId, message, chatId, chatType, messageId, rootId, threadId);
        const ctx: RoutingContext = { chatId, messageId, chatType, larkAppId, ...routing };

        const ownsSession = handlers.isSessionOwner?.(routing.anchor) ?? false;

        // Simple permission gate: if not session owner, require @mention
        if (chatType === 'group' && !ownsSession) {
          const mentioned = message.mentions?.some((m: any) => m.id?.open_id === botOpenId) ?? false;
          if (!mentioned) {
            // Check if solo group
            let isSolo = false;
            try {
              const info = await getChatInfo(chatId);
              isSolo = info.userCount <= 1 && info.botCount <= 1;
            } catch { /* ignore */ }
            if (!isSolo) return;
          }
        }

        const promise = ownsSession
          ? handlers.handleThreadReply(data, ctx)
          : handlers.handleNewTopic(data, ctx);
        promise.catch(err => logger.error(`Message handler error: ${err}`));
      } catch (err) {
        logger.error(`Event handling error: ${err}`);
      }
    },
  });

  const wsClient = new Lark.WSClient({
    appId: larkAppId,
    appSecret: larkAppSecret,
    loggerLevel: process.env.DEBUG ? Lark.LoggerLevel.info : Lark.LoggerLevel.warn,
  });

  wsClient.start({ eventDispatcher });
  return wsClient;
}

async function decideRouting(
  larkAppId: string,
  message: any,
  chatId: string,
  chatType: 'group' | 'p2p',
  messageId: string,
  rootId?: string,
  threadId?: string,
): Promise<{ scope: 'thread' | 'chat'; anchor: string }> {
  // Real thread reply
  if (rootId && threadId) return { scope: 'thread', anchor: rootId };
  // P2P
  if (chatType === 'p2p') return { scope: 'thread', anchor: messageId };
  // Group — check chat_mode
  const mode = await getChatMode(chatId);
  if (mode === 'topic') return { scope: 'thread', anchor: messageId };
  return { scope: 'chat', anchor: chatId };
}

      if (ds?.scope === 'chat') {
        return sendMessage(ds.chatId, content, msgType);
      }
      return replyMessage(anchor, content, msgType, true);
    },
    getAnchorForSession: (key) => {
      const ds = sessionManager.getSession(key);
      return ds ? sessionAnchorId(ds) : '';
    },
  });

  // Card handler deps
  const cardDeps = {
    interactiveBridge,
    onToggleDisplay: (sessionId: string, cardNonce?: string) => {
      // Toggle display mode for session
      const ds = sessionManager.getSession(sessionId);
      if (!ds) return buildStreamingCard({ sessionId, rootId: sessionAnchorId(ds!), title: '', markdownContent: '', status: 'idle', displayMode: 'hidden' });
      const newMode = ds.frozenCards ? 'hidden' : 'markdown';
      return buildStreamingCard({
        sessionId,
        rootId: sessionAnchorId(ds),
        title: ds.currentTurnTitle ?? 'Session',
        markdownContent: ds.lastScreenContent ?? '',
        status: 'idle',
        displayMode: newMode as 'hidden' | 'markdown',
      });
    },
    onRestart: async (sessionId: string) => {
      return JSON.stringify({ text: 'Restart not yet implemented' });
    },
    onClose: async (sessionId: string) => {
      sessionManager.removeSession(sessionId);
      return JSON.stringify({ text: 'Session closed.' });
    },
  };

  // Event handlers
  const handlers = {
    handleNewTopic: async (data: any, ctx: any) => {
      const msg = parseEventMessage(data);
      const ds = await sessionManager.createSession(ctx, msg.content, msg.senderId);

      // Send streaming card
      const card = buildStreamingCard({
        sessionId: ds.session.sessionId,
        rootId: sessionAnchorId(ds),
        title: msg.content.substring(0, 50),
        markdownContent: '',
        status: 'starting',
        displayMode: 'hidden',
      });

      const replyAnchor = ctx.scope === 'chat' ? ctx.chatId : ctx.anchor;
      const msgId = ctx.scope === 'chat'
        ? await sendMessage(ctx.chatId, card, 'interactive')
        : await replyMessage(ctx.anchor, card, 'interactive', true);
      ds.streamCardId = msgId;
      ds.currentTurnTitle = msg.content.substring(0, 50);

      // Start agent turn
      await sessionManager.runAgentTurn(ds, msg.content);
    },

    handleThreadReply: async (data: any, ctx: any) => {
      const msg = parseEventMessage(data);
      const key = sessionKey(ctx.anchor, ctx.larkAppId);
      let ds = sessionManager.getSession(key);

      if (!ds) {
        // Auto-create (safety net)
        ds = await sessionManager.createSession(ctx, msg.content, msg.senderId);
      }

      // Check for daemon commands
      const stripped = stripLeadingMentions(msg.content, msg.mentions);
      if (stripped.startsWith('/')) {
        const [cmd, ...args] = stripped.split(/\s+/);
        if (DAEMON_COMMANDS.has(cmd!)) {
          const anchor = sessionAnchorId(ds);
          const handled = await handleCommand(cmd!, args.join(' '), ds, sessionManager,
            async (a, c, t) => {
              if (ds!.scope === 'chat') return sendMessage(ds!.chatId, c, t);
              return replyMessage(a, c, t, true);
            },
          );
          if (handled) return;
        }
      }

      // Queue or run
      if (ds.busy) {
        sessionManager.queueMessage(ds, msg.content);
      } else {
        // Start new turn → new card
        ds.streamCardPending = true;
        const card = buildStreamingCard({
          sessionId: ds.session.sessionId,
          rootId: sessionAnchorId(ds),
          title: msg.content.substring(0, 50),
          markdownContent: '',
          status: 'working',
          displayMode: 'hidden',
        });
        const anchor = sessionAnchorId(ds);
        const msgId = ds.scope === 'chat'
          ? await sendMessage(ds.chatId, card, 'interactive')
          : await replyMessage(anchor, card, 'interactive', true);
        ds.streamCardId = msgId;
        ds.currentTurnTitle = msg.content.substring(0, 50);

        await sessionManager.runAgentTurn(ds, msg.content);
      }
    },

    handleCardAction: async (data: any) => {
      return handleCardAction(data, cardDeps);
    },

    isSessionOwner: (anchor: string) => {
      return sessionManager.getSession(sessionKey(anchor, larkAppId)) !== undefined;
    },
  };

  // Start WS
  const wsClient = startLarkEventDispatcher(larkAppId, larkAppSecret, handlers, botIdentity.openId);

  // Graceful shutdown
  const shutdown = () => {
    logger.info('Daemon shutting down...');
    wsClient.stop?.();
    process.exit(0);
  };
  process.on('SIGTERM', shutdown);
  process.on('SIGINT', shutdown);

  logger.info(`Daemon started for profile "${profileId}" (${botIdentity.name})`);
}

// Agent event → card update
async function handleAgentEvent(
  key: string,
  event: AgentEvent,
  sessionManager: SessionManager,
): Promise<void> {
  const ds = sessionManager.getSession(key);
  if (!ds || !ds.streamCardId) return;

  switch (event.type) {
    case 'text_delta': {
      ds.lastScreenContent = (ds.lastScreenContent ?? '') + event.delta;
      const card = buildStreamingCard({
        sessionId: ds.session.sessionId,
        rootId: sessionAnchorId(ds),
        title: ds.currentTurnTitle ?? 'Working',
        markdownContent: ds.lastScreenContent,
        status: 'working',
        displayMode: 'markdown',
      });
      if (!ds.cardPatchInFlight) {
        ds.cardPatchInFlight = true;
        updateMessage(ds.streamCardId, card)
          .finally(() => { ds.cardPatchInFlight = false; });
      } else {
        ds.pendingCardJson = card;
      }
      break;
    }
    case 'turn_complete': {
      // Freeze card as idle
      const card = buildStreamingCard({
        sessionId: ds.session.sessionId,
        rootId: sessionAnchorId(ds),
        title: ds.currentTurnTitle ?? 'Done',
        markdownContent: ds.lastScreenContent ?? '',
        status: 'idle',
        displayMode: 'markdown',
      });
      updateMessage(ds.streamCardId, card).catch(() => {});
      break;
    }
  }
}

export const daemonCommands: SlashCommand[] = [
  {
    name: 'daemon',
    description: 'Browse running daemon sessions',
    usage: '/daemon',
    handler: async (args, context) => {
      // List running daemons by scanning PID files
      const { readdirSync, existsSync, readFileSync } = require('node:fs');
      const { join } = require('node:path');
      const { homedir } = require('node:os');
      const daemonDir = join(homedir() ?? '/root', '.my-agent', 'data');
      if (!existsSync(daemonDir)) {
        return 'No daemon data directory found. Start a daemon first with `my-agent daemon start <profile>`.';
      }

      const pidFiles = readdirSync(daemonDir).filter(f => f.endsWith('.pid'));
      if (pidFiles.length === 0) {
        return 'No running daemons found. Start a daemon first with `my-agent daemon start <profile>`.';
      }

      const lines = ['**Running Daemons:**', ''];
      for (const pf of pidFiles) {
        try {
          const pid = readFileSync(join(daemonDir, pf), 'utf-8').trim();
          lines.push(`- ${pf.replace('.pid', '')} (PID: ${pid})`);
        } catch { /* skip */ }
      }
      return lines.join('\n');
    },
  },
];

// CLI command handlers for profile create, bot add, daemon start/stop/list

export async function profileCreate(name: string): Promise<void> {
  // Interactive profile creation
  const { createInterface } = require('node:readline/promises');
  const rl = createInterface({ input: process.stdin, output: process.stdout });

  console.log(`Creating profile "${name}"...`);
  const role = await rl.question('Role description: ');
  const workingDir = await rl.question('Working directory (~/projects/api): ') || '~/projects/api';
  console.log('Tool profile:');
  console.log('  1. read_only (read, grep, glob, ls)');
  console.log('  2. code_editor (+ text_editor, bash)');
  console.log('  3. general (all tools)');
  const toolChoice = await rl.question('Choose [2]: ') || '2';
  const toolProfile = { '1': 'read_only', '2': 'code_editor', '3': 'general' }[toolChoice] ?? 'code_editor';
  const model = await rl.question('Model (enter for default): ') || undefined;
  rl.close();

  // Create workspace
  const { resolvePath, getBotsConfigPath } = await import('../profile/loader');
  const { mkdirSync, writeFileSync, readFileSync, existsSync } = require('node:fs');
  const { join, dirname } = require('node:path');
  const { parse: parseYaml, stringify: stringifyYaml } = require('yaml');

  const wsPath = resolvePath(`~/.my-agent/profiles/${name}`);
  mkdirSync(wsPath, { recursive: true });

  // Create identity files
  writeFileSync(join(wsPath, 'AGENTS.md'), `# ${name}\n\n${role}\n`);
  writeFileSync(join(wsPath, 'SOUL.md'), `# Personality\n\nI am a helpful ${name} agent.\n`);
  writeFileSync(join(wsPath, 'IDENTITY.md'), `# Identity\n\n- Name: ${name}\n- Role: ${role}\n`);

  // Update bots.yml
  const configPath = getBotsConfigPath();
  let config: any = { profiles: {}, bots: [] };
  if (existsSync(configPath)) {
    config = parseYaml(readFileSync(configPath, 'utf-8'));
  }
  config.profiles[name] = {
    workspace: `~/.my-agent/profiles/${name}`,
    toolProfile,
    workingDir,
    ...(model ? { model } : {}),
  };
  mkdirSync(dirname(configPath), { recursive: true });
  writeFileSync(configPath, stringifyYaml(config));

  console.log(`✅ Profile "${name}" created at ${wsPath}`);
}
