# Spec: API 类型 + Provider 扩展机制

## Problem

当前 `@my-agent-team/ai` 包的 `anthropicProvider` 内部 `new AnthropicChatModel` 直接依赖 `@anthropic-ai/sdk`。加新 provider（OpenAI/DeepSeek/Groq）要写新的 ChatModel 实现。没有 API 类型概念，无法复用 API 实现。

## Goal

参考 pi-ai 的 API 层 + Provider 层分离设计：

1. 定义 `Api` 类型概念（`"anthropic-messages"` | `"openai-completions"` | string）
2. 每个 API 实现导出 `stream(model, messages, options) -> AsyncIterable<AIMessageChunk>`
3. Provider 配置声明用哪个 `api`，多个 provider 可复用同一 API 实现
4. 加 DeepSeek/Groq/xAI 等 OpenAI 兼容 provider 只需 5 行配置
5. 支持自定义 baseUrl 的 OpenAI 兼容 provider

## Design

### 类型定义

```typescript
// packages/ai/src/types.ts (扩展)

/** API 类型标识。已知类型有编译时提示，自定义类型通过 string 扩展。 */
export type KnownApi = "anthropic-messages" | "openai-completions";
export type Api = KnownApi | (string & {});

/** Model 增加 api 字段 */
export interface Model {
  id: string;
  name: string;
  provider: string;
  api: Api;               // 新增：声明此模型用哪个 API 实现
  baseUrl?: string;
  reasoning: boolean;
  input: InputModality[];
  cost: ModelCost;
  contextWindow: number;
  maxTokens: number;
}

/** API 流选项 */
export interface ApiStreamOptions {
  apiKey?: string;
  baseUrl?: string;
  headers?: Record<string, string>;
  signal?: AbortSignal;
  tools?: readonly Tool[];
}

/** API 实现接口 */
export interface ApiImplementation {
  stream(
    model: Model,
    messages: readonly Message[],
    options?: ApiStreamOptions,
  ): AsyncIterable<AIMessageChunk>;
}
```

### API 层

```typescript
// packages/ai/src/api/anthropic-messages.ts

export const anthropicMessagesApi: ApiImplementation = {
  async *stream(model, messages, options) {
    // 直接用 fetch 调 Anthropic Messages API
    // 解析 SSE 流，yield AIMessageChunk
    const response = await fetch(`${options?.baseUrl ?? "https://api.anthropic.com"}/v1/messages`, {
      method: "POST",
      headers: {
        "x-api-key": options?.apiKey ?? "",
        "anthropic-version": "2023-06-01",
        "content-type": "application/json",
        ...options?.headers,
      },
      body: JSON.stringify({
        model: model.id,
        max_tokens: model.maxTokens,
        messages: convertMessages(messages),
        ...(options?.tools?.length ? { tools: convertTools(options.tools) } : {}),
        stream: true,
      }),
      signal: options?.signal,
    });

    // 解析 SSE 流
    for await (const event of parseSSE(response.body!)) {
      // event.type: message_start / content_block_start / content_block_delta / message_delta / message_stop
      // 转换为 AIMessageChunk
      if (event.type === "content_block_delta") {
        if (event.delta.type === "text_delta") {
          yield { delta: { type: "text", text: event.delta.text } };
        } else if (event.delta.type === "input_json_delta") {
          yield { delta: { type: "input_json_delta", id: event.index, partial_json: event.delta.partial_json } };
        } else if (event.delta.type === "thinking_delta") {
          yield { delta: { type: "reasoning", text: event.delta.thinking } };
        }
      } else if (event.type === "message_start") {
        if (event.message.usage) {
          yield {
            usage: {
              input: event.message.usage.input_tokens,
              output: event.message.usage.output_tokens,
              cacheCreate: event.message.usage.cache_creation_input_tokens,
              cacheRead: event.message.usage.cache_read_input_tokens,
            },
          };
        }
      } else if (event.type === "message_delta") {
        if (event.delta.stop_reason) {
          yield { stopReason: mapStopReason(event.delta.stop_reason) };
        }
        if (event.usage) {
          yield {
            usage: {
              input: 0,
              output: event.usage.output_tokens,
            },
          };
        }
      } else if (event.type === "message_stop") {
        yield { done: true };
      }
    }
  },
};
```

```typescript
// packages/ai/src/api/openai-completions.ts

export const openAICompletionsApi: ApiImplementation = {
  async *stream(model, messages, options) {
    // 用 fetch 调 OpenAI Chat Completions API
    // 兼容 DeepSeek/Groq/xAI 等 OpenAI 兼容 API
    const response = await fetch(`${options?.baseUrl ?? "https://api.openai.com"}/v1/chat/completions`, {
      method: "POST",
      headers: {
        "authorization": `Bearer ${options?.apiKey ?? ""}`,
        "content-type": "application/json",
        ...options?.headers,
      },
      body: JSON.stringify({
        model: model.id,
        max_tokens: model.maxTokens,
        messages: convertMessages(messages),
        ...(options?.tools?.length ? { tools: convertTools(options.tools) } : {}),
        stream: true,
      }),
      signal: options?.signal,
    });

    // 解析 SSE 流 (data: {...}\n\n)
    for await (const chunk of parseOpenAISSE(response.body!)) {
      if (chunk.choices?.[0]?.delta?.content) {
        yield { delta: { type: "text", text: chunk.choices[0].delta.content } };
      }
      if (chunk.choices?.[0]?.delta?.tool_calls) {
        for (const tc of chunk.choices[0].delta.tool_calls) {
          if (tc.function?.arguments) {
            yield { delta: { type: "input_json_delta", id: tc.id, partial_json: tc.function.arguments } };
          }
          if (tc.id) {
            yield { delta: { type: "tool_use", id: tc.id, name: tc.function?.name ?? "" } };
          }
        }
      }
      if (chunk.choices?.[0]?.finish_reason) {
        yield { stopReason: mapStopReason(chunk.choices[0].finish_reason) };
      }
      if (chunk.usage) {
        yield {
          usage: {
            input: chunk.usage.prompt_tokens,
            output: chunk.usage.completion_tokens,
          },
        };
      }
    }
    yield { done: true };
  },
};
```

### Provider 层

```typescript
// packages/ai/src/providers/anthropic.ts

export function anthropicProvider(auth: ProviderAuth): Provider {
  const cache = new Map<string, ChatModel>();
  return {
    id: "anthropic",
    name: "Anthropic",
    baseUrl: auth.baseUrl,
    getModels: () => ANTHROPIC_MODELS,  // 每个 Model 有 api: "anthropic-messages"
    createModel(model: Model, opts?: ProviderAuth): ChatModel {
      // 从 api registry 获取实现
      const apiImpl = apiRegistry.get(model.api);
      if (!apiImpl) throw new Error(`Unknown API: ${model.api}`);
      return {
        id: model.id,
        async *stream(messages, options) {
          yield* apiImpl.stream(model, messages, {
            apiKey: opts?.apiKey ?? auth.apiKey,
            baseUrl: opts?.baseUrl ?? auth.baseUrl,
            headers: opts?.headers,
            signal: options?.signal,
            tools: options?.tools,
          });
        },
      };
    },
  };
}
```

### API Registry

```typescript
// packages/ai/src/api/registry.ts

const registry = new Map<Api, ApiImplementation>();

export function registerApi(api: Api, impl: ApiImplementation): void {
  registry.set(api, impl);
}

export function getApiImplementation(api: Api): ApiImplementation | undefined {
  return registry.get(api);
}

// 启动时注册
registerApi("anthropic-messages", anthropicMessagesApi);
registerApi("openai-completions", openAICompletionsApi);
```

### createOpenAICompatProvider 工厂

```typescript
// packages/ai/src/providers/openai-compat.ts

export function createOpenAICompatProvider(config: {
  id: string;
  name: string;
  baseUrl: string;
  auth: ProviderAuth;
  models: Model[];
}): Provider {
  return {
    id: config.id,
    name: config.name,
    baseUrl: config.baseUrl,
    getModels: () => config.models,
    createModel(model: Model, opts?: ProviderAuth): ChatModel {
      return {
        id: model.id,
        async *stream(messages, options) {
          const impl = getApiImplementation(model.api);
          if (!impl) throw new Error(`Unknown API: ${model.api}`);
          yield* impl.stream(model, messages, {
            apiKey: opts?.apiKey ?? config.auth.apiKey,
            baseUrl: opts?.baseUrl ?? config.baseUrl,
            signal: options?.signal,
            tools: options?.tools,
          });
        },
      };
    },
  };
}
```

### SSE 解析工具

```typescript
// packages/ai/src/api/sse-parser.ts

/** 解析 SSE 流，yield 每个事件的 JSON data。 */
export async function* parseSSE(
  body: ReadableStream<Uint8Array>,
): AsyncIterable<Record<string, unknown>> {
  const reader = body.getReader();
  const decoder = new TextDecoder();
  let buffer = "";
  try {
    while (true) {
      const { done, value } = await reader.read();
      if (done) break;
      buffer += decoder.decode(value, { stream: true });
      const lines = buffer.split("\n");
      buffer = lines.pop() ?? "";
      for (const line of lines) {
        if (line.startsWith("data: ")) {
          const data = line.slice(6);
          if (data === "[DONE]") return;
          try {
            yield JSON.parse(data);
          } catch {
            // skip malformed
          }
        }
      }
    }
  } finally {
    reader.releaseLock();
  }
}
```

## 消息转换

### Anthropic 格式

```typescript
function convertMessages(messages: readonly Message[]): unknown[] {
  // 我们的 Message -> Anthropic MessageParam[]
  // system messages 合并为 top-level system 参数（在 body 里，不在 messages 里）
  // user messages -> { role: "user", content: string | content[] }
  // assistant messages -> { role: "assistant", content: ContentBlock[] }
  // tool results -> { role: "user", content: [{ type: "tool_result", ... }] }
}
```

### OpenAI 格式

```typescript
function convertMessagesOpenAI(messages: readonly Message[]): unknown[] {
  // 我们的 Message -> OpenAI ChatCompletionMessageParam[]

  // user -> { role: "user", content: string }
  // assistant -> { role: "assistant", content: string, tool_calls?: [...] }
  // tool results -> { role: "tool", tool_call_id: ..., content: ... }
}
```

## 不做

- 不做 OAuth（API key 足够）
- 不做 OpenAI Responses API（Chat Completions 兼容性更广）
- 不做 Google/Bedrock API（先做 Anthropic + OpenAI 两大主流）
- 不做图片输入（先支持纯文本 + tool_use）
- 不做 prompt cache 控制（API 层不处理 cache_control）
- 不做 stealth mode（不伪装 Claude Code）
- 不做 thinking level 映射（先支持 basic reasoning）
